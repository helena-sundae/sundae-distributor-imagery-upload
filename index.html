function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var imageLinks = []; if (data.files && data.files.length > 0) { var rootFolder = DriveApp.getFolderById('1V25aTfceGmLRvoAIBQoKDxa1qR4WsDxk'); var countryFolders = rootFolder.getFoldersByName(data.country); var countryFolder = countryFolders.hasNext() ? countryFolders.next() : rootFolder.createFolder(data.country); var isMultiple = data.files.length > 1; var targetFolder = countryFolder; if (isMultiple) { var subFolders = countryFolder.getFoldersByName(data.prefix); targetFolder = subFolders.hasNext() ? subFolders.next() : countryFolder.createFolder(data.prefix); } data.files.forEach(function(file) { var blob = Utilities.newBlob(Utilities.base64Decode(file.data), file.type, file.name); var uploaded = targetFolder.createFile(blob); uploaded.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); if (!isMultiple) imageLinks.push(uploaded.getUrl()); }); if (isMultiple) { targetFolder.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); imageLinks.push(targetFolder.getUrl()); } } sheet.appendRow([ data.dateSubmitted, data.country, data.retailer, data.storeLocation, data.datePhotoTaken, data.imageType, data.secondaryPlacementType, data.products, data.comments, data.filesUploaded, imageLinks.join(', ') ]); highlightNewRow(); updateDashboard(); return ContentService.createTextOutput(JSON.stringify({ result: 'success', links: imageLinks })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ result: 'error', error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({ result: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } function highlightNewRow() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var lastRow = sheet.getLastRow(); var range = sheet.getRange(lastRow, 1, 1, sheet.getLastColumn()); range.setBackground('#fff3cd'); } function updateDashboard() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var dataSheet = ss.getSheets()[0]; var dashSheet = ss.getSheetByName('Dashboard'); var data = dataSheet.getDataRange().getValues(); dashSheet.clearContents(); var charts = dashSheet.getCharts(); charts.forEach(function(chart) { dashSheet.removeChart(chart); }); var total = data.length - 1; var countries = {}, retailers = {}, imageTypes = {}, products = {}; for (var i = 1; i < data.length; i++) { var country = data[i][1].toString().trim().toLowerCase().replace(/\b\w/g, function(l){ return l.toUpperCase(); }); var retailer = data[i][2].toString().trim().toLowerCase().replace(/\b\w/g, function(l){ return l.toUpperCase(); }); var imageType = data[i][5].toString().trim().toLowerCase().replace(/\b\w/g, function(l){ return l.toUpperCase(); }); var productList = data[i][7] ? data[i][7].toString().split(', ') : []; countries[country] = (countries[country] || 0) + 1; retailers[retailer] = (retailers[retailer] || 0) + 1; imageTypes[imageType] = (imageTypes[imageType] || 0) + 1; productList.forEach(function(p) { if (p) { var normP = p.trim().toLowerCase().replace(/\b\w/g, function(l){ return l.toUpperCase(); }); products[normP] = (products[normP] || 0) + 1; } }); } dashSheet.getRange('A1').setValue('SUNDAE IMAGERY DASHBOARD'); dashSheet.getRange('A2').setValue('Last updated: ' + new Date().toLocaleString()); dashSheet.getRange('A4').setValue('Total submissions'); dashSheet.getRange('B4').setValue(total); dashSheet.getRange('A6').setValue('BY COUNTRY'); var countryRow = 7; for (var c in countries) { dashSheet.getRange(countryRow, 1).setValue(c); dashSheet.getRange(countryRow, 2).setValue(countries[c]); countryRow++; } dashSheet.getRange('D6').setValue('BY RETAILER'); var retailerRow = 7; for (var r in retailers) { dashSheet.getRange(retailerRow, 4).setValue(r); dashSheet.getRange(retailerRow, 5).setValue(retailers[r]); retailerRow++; } dashSheet.getRange('A' + (countryRow + 2)).setValue('BY IMAGE TYPE'); var imgRow = countryRow + 3; for (var t in imageTypes) { dashSheet.getRange(imgRow, 1).setValue(t); dashSheet.getRange(imgRow, 2).setValue(imageTypes[t]); imgRow++; } dashSheet.getRange('D' + (imgRow + 2)).setValue('BY PRODUCT'); var prodRow = imgRow + 3; for (var p in products) { dashSheet.getRange(prodRow, 4).setValue(p); dashSheet.getRange(prodRow, 5).setValue(products[p]); prodRow++; } var countryRange = dashSheet.getRange(6, 1, countryRow - 6, 2); var countryChart = dashSheet.newChart() .setChartType(Charts.ChartType.PIE) .addRange(countryRange) .setPosition(6, 7, 0, 0) .setOption('title', 'Submissions by Country') .setOption('width', 400) .setOption('height', 300) .build(); dashSheet.insertChart(countryChart); var retailerRange = dashSheet.getRange(6, 4, retailerRow - 6, 2); var retailerChart = dashSheet.newChart() .setChartType(Charts.ChartType.BAR) .addRange(retailerRange) .setPosition(18, 7, 0, 0) .setOption('title', 'Submissions by Retailer') .setOption('width', 400) .setOption('height', 300) .build(); dashSheet.insertChart(retailerChart); var imgTypeRange = dashSheet.getRange(countryRow + 2, 1, imgRow - countryRow - 2, 2); var imgChart = dashSheet.newChart() .setChartType(Charts.ChartType.BAR) .addRange(imgTypeRange) .setPosition(6, 13, 0, 0) .setOption('title', 'Submissions by Image Type') .setOption('width', 400) .setOption('height', 300) .build(); dashSheet.insertChart(imgChart); var productRange = dashSheet.getRange(imgRow + 2, 4, prodRow - imgRow - 2, 2); var productChart = dashSheet.newChart() .setChartType(Charts.ChartType.BAR) .addRange(productRange) .setPosition(18, 13, 0, 0) .setOption('title', 'Submissions by Product') .setOption('width', 400) .setOption('height', 300) .build(); dashSheet.insertChart(productChart); }
